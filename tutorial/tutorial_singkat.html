5. üèóÔ∏è Model & Autentikasi

6Model User

File: models/User.php

<?php class User { private $conn; private $table_name = "users"; public function __construct($db) { $this->conn = $db; } public function login($username, $password) { $query = "SELECT id, username, password, nama_lengkap, role FROM " . $this->table_name . " WHERE username = :username"; $stmt = $this->conn->prepare($query); $stmt->bindParam(':username', $username); $stmt->execute(); if ($stmt->rowCount() > 0) { $row = $stmt->fetch(PDO::FETCH_ASSOC); if (password_verify($password, $row['password'])) { return $row; } } return false; } } ?>
